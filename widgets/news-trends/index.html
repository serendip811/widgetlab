코드

// === 실시간 인기 뉴스 위젯 ===
// Google News 한국 헤드라인 5개
// 탭하면 해당 뉴스로 이동

const LIMIT = 5
const RSS_URL = "https://news.google.com/rss?hl=ko&gl=KR&ceid=KR:ko"

const w = new ListWidget()
w.setPadding(14, 14, 14, 14)
w.backgroundColor = new Color("#111827")

const title = w.addText("🔥 지금 많이 보는 뉴스")
title.font = Font.boldSystemFont(15)
title.textColor = Color.white()

w.addSpacer(8)

try {
  const req = new Request(RSS_URL)
  req.headers = {
    "User-Agent": "Mozilla/5.0"
  }

  const xml = await req.loadString()
  const items = parseItems(xml).slice(0, LIMIT)

  for (let i = 0; i < items.length; i++) {
    const item = items[i]

    const row = w.addStack()
    row.layoutHorizontally()
    row.centerAlignContent()
    row.url = item.link

    const rank = row.addText(`${i + 1}`)
    rank.font = Font.boldSystemFont(13)
    rank.textColor = new Color("#FBBF24")

    row.addSpacer(8)

    const text = row.addText(cleanTitle(item.title))
    text.font = Font.systemFont(13)
    text.textColor = Color.white()
    text.lineLimit = 2

    w.addSpacer(6)
  }

  w.addSpacer(4)

  const foot = w.addText("Google News 기준")
  foot.font = Font.systemFont(10)
  foot.textColor = new Color("#9CA3AF")

} catch (e) {
  const err = w.addText("뉴스를 불러오지 못했어요")
  err.textColor = Color.white()
  err.font = Font.boldSystemFont(13)

  w.addSpacer(4)

  const msg = w.addText(String(e))
  msg.textColor = new Color("#9CA3AF")
  msg.font = Font.systemFont(10)
}

Script.setWidget(w)
Script.complete()

function parseItems(xml) {
  const blocks = xml.match(/[\s\S]*?<\/item>/g) || []
  const items = []

  for (const block of blocks) {
    const title = decode(getTag(block, "title"))
    const link = decode(getTag(block, "link"))

    if (title && link) {
      items.push({ title, link })
    }
  }

  return items
}

function getTag(text, tag) {
  const re = new RegExp(`<${tag}[^>]*>([\\s\\S]*?)<\\/${tag}>`)
  const m = text.match(re)
  return m ? m[1].trim() : ""
}

function decode(str) {
  return str
    .replace(//g, "")
    .replace(/&/g, "&")
    .replace(/"/g, '"')
    .replace(/'/g, "'")
    .replace(/</g, "<")
    .replace(/>/g, ">")
}

function cleanTitle(title) {
  return title.replace(/\s-\s[^-]+$/g, "")
}
