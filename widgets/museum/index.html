코드

// 위젯연구소 - 어린이박물관 위젯 샘플
// 실제 예약 상태 확인 로직으로 message 값을 교체하세요.

const title = "어린이박물관";
const message = "예약 페이지 확인";
const subcopy = "방문 날짜와 회차를 확인하세요.";
const link = "https://www.museum.go.kr";

const widget = new ListWidget();
widget.backgroundColor = new Color("#fff8f2");
widget.url = link;
widget.setPadding(18, 18, 18, 18);

const titleText = widget.addText(title);
titleText.font = Font.boldSystemFont(17);
titleText.textColor = new Color("#362719");

widget.addSpacer(12);

const messageText = widget.addText(message);
messageText.font = Font.semiboldSystemFont(23);
messageText.textColor = new Color("#b85d32");
messageText.minimumScaleFactor = 0.7;

widget.addSpacer(8);

const subcopyText = widget.addText(subcopy);
subcopyText.font = Font.systemFont(13);
subcopyText.textColor = new Color("#6f5847");
subcopyText.lineLimit = 2;

widget.addSpacer();

const footer = widget.addText("눌러서 예약 페이지 열기");
footer.font = Font.mediumSystemFont(12);
footer.textColor = new Color("#21705b");

Script.setWidget(widget);
Script.complete();
