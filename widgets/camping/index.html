// 위젯연구소 - 서울대공원 캠핑장 위젯 샘플
// 실제 예약 확인 API 또는 웹 파싱 로직으로 status 값을 교체하세요.

const title = "서울대공원 캠핑장";
const status = "예약 페이지 확인 필요";
const detail = "캠핑 일정이 열리는 시간에 맞춰 확인하세요.";
const link = "https://grandpark.seoul.go.kr";

const widget = new ListWidget();
widget.backgroundColor = new Color("#f6fbf7");
widget.url = link;
widget.setPadding(18, 18, 18, 18);

const heading = widget.addText(title);
heading.font = Font.boldSystemFont(17);
heading.textColor = new Color("#17352b");

widget.addSpacer(10);

const statusText = widget.addText(status);
statusText.font = Font.semiboldSystemFont(24);
statusText.textColor = new Color("#0f7a5d");
statusText.minimumScaleFactor = 0.7;

widget.addSpacer(8);

const detailText = widget.addText(detail);
detailText.font = Font.systemFont(13);
detailText.textColor = new Color("#45645a");
detailText.lineLimit = 2;

widget.addSpacer();

const footer = widget.addText("눌러서 예약 페이지 열기");
footer.font = Font.mediumSystemFont(12);
footer.textColor = new Color("#cf6f45");

Script.setWidget(widget);
Script.complete();
