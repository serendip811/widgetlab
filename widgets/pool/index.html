코드

// 위젯연구소 - 수영장 예약 위젯 샘플
// 실제 시설별 예약 확인 로직으로 values를 교체하세요.

const values = {
  title: "수영장 예약",
  status: "예약 페이지 확인",
  detail: "원하는 시간대가 열렸는지 확인하세요.",
  link: "https://yeyak.seoul.go.kr"
};

const widget = new ListWidget();
widget.backgroundColor = new Color("#f3f8ff");
widget.url = values.link;
widget.setPadding(18, 18, 18, 18);

const title = widget.addText(values.title);
title.font = Font.boldSystemFont(17);
title.textColor = new Color("#172b45");

widget.addSpacer(12);

const status = widget.addText(values.status);
status.font = Font.semiboldSystemFont(23);
status.textColor = new Color("#2767b1");
status.minimumScaleFactor = 0.7;

widget.addSpacer(8);

const detail = widget.addText(values.detail);
detail.font = Font.systemFont(13);
detail.textColor = new Color("#4a5e78");
detail.lineLimit = 2;

widget.addSpacer();

const footer = widget.addText("눌러서 예약 페이지 열기");
footer.font = Font.mediumSystemFont(12);
footer.textColor = new Color("#c66b43");

Script.setWidget(widget);
Script.complete();
